# Make several requests
curl http://$API_IP/api/test

View application container logs:

# Get one of the pod names
export POD_NAME=$(kubectl get pods -l app=api-gateway -o jsonpath='{.items[0].metadata.name}')

# View main container logs (should see Flask startup messages)
kubectl logs $POD_NAME -c api-gateway



View Fluent Bit sidecar logs:

# View sidecar container logs (should see log entries from the file)
kubectl logs $POD_NAME -c fluent-bit





kubectl apply -f services.yaml
kubectl get svc



# Get service IPs
export API_IP=$(kubectl get svc api-gateway -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
export ORDER_IP=$(kubectl get svc order-service -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
export PAYMENT_IP=$(kubectl get svc payment-service -o jsonpath='{.status.loadBalancer.ingress[0].ip}')

# Generate requests
for i in {1..5}; do
  curl http://$API_IP/api/test
  curl http://$ORDER_IP/orders
  curl http://$PAYMENT_IP/payments
  sleep 1
done





# API Gateway logs
kubectl logs -l app=api-gateway -c fluent-bit --tail=20

# Order Service logs
kubectl logs -l app=order-service -c fluent-bit --tail=20

# Payment Service logs
kubectl logs -l app=payment-service -c fluent-bit --tail=20
